from statsmodels.stats.weightstats import ztest as ztest

alpha  = 0.05

z_stat, p_value = ztest(x_train["age"], x_test["age"], value=0) 

print("The Z-statistic is", z_stat, "while the p_value is", p_value, "\n")

if(p_value < alpha):
  print("Since the p-value",p_value,"is less than 0.05, we have sufficient evidence to reject the null hypothesis.")
else:
  print("Since the p-value",p_value,"is not less than 0.05, we do not have sufficient evidence to reject the null hypothesis.")
